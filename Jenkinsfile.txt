pipeline {
  agent any
  options {
    buildDiscarder(logRotator(numToKeepStr: '5'))
  }
  environment {
    IMAGE_NAME = 'manundi/spineDemo'
    IMAGE_TAG = 'latest'
    APP_NAME = 'spineDemo'
  }
  stages {
    stage('Build') {
      steps {
        sh 'docker build -t $IMAGE_NAME:$IMAGE_TAG .'
      }
    }
  }
  post {
    always {
      sh 'docker logout'
    }
  }
}
